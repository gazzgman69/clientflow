function listRoutes(app:any){

  app._router?.stack?.forEach((mw:any)=>{
    if (mw.route){
      routes.push({method:Object.keys(mw.route.methods)[0], path:mw.route.path});
    } else if (mw.name==="router" && mw.handle?.stack){

}
