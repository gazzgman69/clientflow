4) Route ordering dump (ADD + PRINT)

Add this helper and call it after routes are registered (only if DEBUG_OAUTH=1):

function listRoutes(app:any){
  const out:any[]=[];
  const stack = app?._router?.stack || [];
  for (const mw of stack){
    if (mw.route){
      out.push({ method:Object.keys(mw.route.methods)[0], path:mw.route.path });
    } else if (mw.name==='router' && mw.handle?.stack){
      mw.handle.stack.forEach((h:any)=>{
        if (h.route) out.push({ method:Object.keys(h.route.methods)[0], path:h.route.path });
      });
    }
  }
  console.info('[ROUTES]', JSON.stringify(out, null, 2));
}
if (process.env.DEBUG_OAUTH === '1') listRoutes(app);

The full [ROUTES] printout from server logs.
