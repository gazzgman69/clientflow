Part 1 — Implement the tiniest read-layer shim

File to edit: server/src/routes/lead-forms.ts (the public submit handler)

Add this helper right after the form is loaded and you have questions (with id and orderIndex):

function normalizeKeys(data: any, questions: any[]) {
  const byIndex = new Map<string, string>(); // "1" -> "q1", "2" -> "q2"
  questions
    .slice()
    .sort((a, b) => (a.orderIndex ?? 0) - (b.orderIndex ?? 0))
    .forEach((q, i) => byIndex.set(String(i + 1), q.id)); // q.id can be "q1" or "q_123..."
  const out: any = {};
  for (const [k, v] of Object.entries(data || {})) {
    const qid = byIndex.get(k) || k; // translate numeric; keep real q_* ids
    out[qid] = v;
  }
  return out;
}
