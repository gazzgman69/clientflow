#!/usr/bin/env tsx
/**
 * End-to-End Lead → Calendar Event Verification (read-only)
 * - Tests numeric & q-style payloads
 * - Verifies event creation in DB
 * - Checks idempotency (no duplicate event on repeat submit)
 * - Prints PASS/FAIL summary
 */

import { neon } from '@neondatabase/serverless';

const DATABASE_URL = process.env.DATABASE_URL!;
const BASE_URL = process.env.BASE_URL || 'http://localhost:5000';
const LCF_SLUG = process.env.LCF_SLUG;

if (!DATABASE_URL) {
  console.error('❌ DATABASE_URL is required');
  process.exit(1);
}
const sql = neon(DATABASE_URL);

type Result = { check: string; status: '✅ PASS'|'❌ FAIL'|'⚠️ INFO'; details: string; };
const results: Result[] = [];

type Question = {
  id: string;
  label: string;
  mapTo?: string;
  orderIndex: number;
  required?: boolean;
};

const nowTag = () => Date.now().toString();

async function getSlug() {
  if (LCF_SLUG) return LCF_SLUG;
  const r = await sql`SELECT slug FROM lead_capture_forms WHERE is_active = true LIMIT 1`;
  if (!r.length) throw new Error('No active lead capture forms found');
  return r[0].slug as string;
}

async function getQuestions(slug: string): Promise<Question[]> {
  const r = await sql`SELECT questions FROM lead_capture_forms WHERE slug=${slug} LIMIT 1`;
  if (!r.length) throw new Error(`No form found for slug ${slug}`);
  const raw = r[0].questions as any;
  const qs: Question[] = typeof raw === 'string' ? JSON.parse(raw) : raw;
  return qs.sort((a,b)=> (a.orderIndex ?? 0) - (b.orderIndex ?? 0));
}

function makeValuesFor(qs: Question[], dateStr: string) {
  const vals: Record<string,string> = {};
  for (const q of qs) {
    const need = q.required ?? false;
    if (!need) continue;
    const map = q.mapTo || '';
    if (/leadName/i.test(map)) vals[q.id] = 'E2E Calendar Triage';
    else if (/leadEmail/i.test(map)) vals[q.id] = `triage+${nowTag()}@example.com`;
    else if (/leadPhoneNumber/i.test(map)) vals[q.id] = '+44 7700 900123';
    else if (/eventType/i.test(map)) vals[q.id] = 'Wedding';
    else if (/eventLocation/i.test(map)) vals[q.id] = 'Test Venue';
    else if (/projectDate/i.test(map)) vals[q.id] = dateStr; // dd/mm/yyyy
    else vals[q.id] = 'N/A';
  }
  return vals;
}

function toNumericKeys(qs: Question[], valuesById: Record<string,string>) {
  const out: Record<string,string> = {};
  qs.forEach((q, idx) => {
    const k = String(idx+1); // 1-based
    if (valuesById[q.id] !== undefined) out[k] = valuesById[q.id];
  });
  return out;
}

async function submit(slug: string, data: Record<string,string>) {
  const res = await fetch(`${BASE_URL}/api/leads/public/${slug}/submit`, {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({ data, consent: true })
  });
  let body: any = null;
  try { body = await res.json(); } catch {}
  return { status: res.status, body };
}

async function listRecentEvents(limit=10) {
  const rows = await sql`
    SELECT title, type,
           to_char(start_date,'DD/MM/YYYY HH24:MI') AS start,
           to_char(created_at,'DD/MM/YYYY HH24:MI') AS created
    FROM events
    ORDER BY created_at DESC
    LIMIT ${limit}
  `;
  return rows as any[];
}

async function countEventsByTitleAndDate(titleLike: string, ddmmyyyy: string) {
  const rows = await sql`
    SELECT count(*)::int AS c
    FROM events
    WHERE title LIKE ${'%' + titleLike + '%'}
      AND to_char(start_date,'DD/MM/YYYY') = ${ddmmyyyy}
  `;
  return rows[0].c as number;
}

(async () => {
  console.log('═══════════════════════════════════════════════════════════════');
  console.log('  END-TO-END LEAD → CALENDAR EVENT VERIFICATION (read-only)');
  console.log('═══════════════════════════════════════════════════════════════\n');

  const slug = await getSlug();
  console.log(`Form slug: ${slug}`);

  const qs = await getQuestions(slug);
  console.log(`Questions loaded: ${qs.length}`);

  // Build values for all REQUIRED fields
  const DATE = '25/12/2025'; // dd/mm/yyyy
  const valuesById = makeValuesFor(qs, DATE);

  // Payloads
  const numericPayload = toNumericKeys(qs, valuesById);     // {"1":...,"2":...}
  const qStylePayload  = valuesById;                         // {"q1":...,"q_...":...}

  // Show what we'll submit
  console.log('\nSubmitting Numeric payload:', JSON.stringify(numericPayload));
  const resNum = await submit(slug, numericPayload);
  console.log('Numeric submit →', resNum.status, resNum.body);

  // Verify numeric → event exists
  const titleLike = 'New Lead Project • E2E Calendar Triage';
  let beforeQ = await countEventsByTitleAndDate(titleLike, DATE);
  // If 400, mark fail now
  if (resNum.status >= 400) {
    results.push({ check: 'A) Numeric → Event', status: '❌ FAIL', details: `HTTP ${resNum.status}: ${JSON.stringify(resNum.body)}` });
  } else {
    // small grace: fetch after submit
    const afterNum = await countEventsByTitleAndDate(titleLike, DATE);
    results.push({
      check: 'A) Numeric → Event',
      status: afterNum > beforeQ ? '✅ PASS' : '❌ FAIL',
      details: `events before=${beforeQ}, after=${afterNum}`
    });
  }

  console.log('\nSubmitting Q-style payload:', JSON.stringify(qStylePayload));
  const resQ = await submit(slug, qStylePayload);
  console.log('Q-style submit →', resQ.status, resQ.body);

  beforeQ = await countEventsByTitleAndDate(titleLike, DATE);
  if (resQ.status >= 400) {
    results.push({ check: 'B) Q-style → Event', status: '❌ FAIL', details: `HTTP ${resQ.status}: ${JSON.stringify(resQ.body)}` });
  } else {
    const afterQ = await countEventsByTitleAndDate(titleLike, DATE);
    results.push({
      check: 'B) Q-style → Event',
      status: afterQ > beforeQ ? '✅ PASS' : '❌ FAIL',
      details: `events before=${beforeQ}, after=${afterQ}`
    });
  }

  // Idempotency: submit same numeric again
  const beforeDup = await countEventsByTitleAndDate(titleLike, DATE);
  const resDup = await submit(slug, numericPayload);
  console.log('\nDuplicate numeric submit →', resDup.status, resDup.body);
  const afterDup = await countEventsByTitleAndDate(titleLike, DATE);
  results.push({
    check: 'C) Idempotency (duplicate submit ≤5m → no second event)',
    status: (afterDup - beforeDup) <= 1 ? '✅ PASS' : '❌ FAIL',
    details: `events before=${beforeDup}, after=${afterDup}`
  });

  // Print last 10 events for visibility
  console.log('\nLast 10 events (dd/mm/yyyy HH:mm):');
  const rows = await listRecentEvents();
  for (const r of rows) {
    console.log(`- ${r.title} | ${r.type} | ${r.start} | ${r.created}`);
  }

  // Summary table
  console.log('\n════════ RESULTS ════════');
  let fails = 0;
  for (const r of results) {
    if (r.status.startsWith('❌')) fails++;
    console.log(`${r.check.padEnd(50)} ${r.status} — ${r.details}`);
  }
  if (fails > 0) {
    console.log(`\n❌ ${fails} check(s) failed`);
    process.exit(1);
  } else {
    console.log('\n✅ All critical checks passed');
  }
})().catch(e => {
  console.error('❌ Script error:', e);
  process.exit(1);
});
