#!/usr/bin/env tsx
import { neon } from '@neondatabase/serverless';
const sql = neon(process.env.DATABASE_URL!);
const TENANT = process.env.TENANT_ID || 'default-tenant';
const PROVIDER = 'google';
const THREAD = 'triage-thread-' + Date.now();

async function main() {
  // create project
  const proj = await sql(`INSERT INTO projects (id, tenant_id, name) VALUES (gen_random_uuid(), $1, 'TEMP PROJ') RETURNING id`, [TENANT]);
  const projectId = proj[0].id;

  // seed thread attached to project
  await sql(`INSERT INTO email_threads (id, tenant_id, provider, thread_id, project_id)
             VALUES (gen_random_uuid(), $1, $2, $3, $4)`, [TENANT, PROVIDER, THREAD, projectId]);

  // delete project (should set thread.project_id = NULL on cascade, after migration)
  await sql(`DELETE FROM projects WHERE id=$1 AND tenant_id=$2`, [projectId, TENANT]);

  // simulate sync re-upsert the same thread (should not fail, should keep NULL project)
  -- this step just reads the row to ensure we didn’t violate FK; if you have a sync function, call it instead

  const rows = await sql(`SELECT thread_id, project_id FROM email_threads WHERE tenant_id=$1 AND provider=$2 AND thread_id=$3`, [TENANT, PROVIDER, THREAD]);
  if (rows.length === 1 && rows[0].project_id === null) {
    console.log('✅ PASS: thread present and detached (project_id IS NULL)');
    process.exit(0);
  } else {
    console.error('❌ FAIL: expected row with project_id NULL, got:', rows);
    process.exit(1);
  }
}
main().catch(e => { console.error('❌ FAIL:', e); process.exit(1); });
