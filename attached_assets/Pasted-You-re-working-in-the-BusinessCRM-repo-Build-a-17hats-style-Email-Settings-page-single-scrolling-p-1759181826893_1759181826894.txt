import { pgTable, text, boolean, integer, timestamp, uuid } from "drizzle-orm/pg-core";

SQL migration (if you prefer raw SQL):

CREATE TABLE IF NOT EXISTS email_providers (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  slug text UNIQUE NOT NULL,
  display_name text NOT NULL,
  supports_incoming boolean NOT NULL DEFAULT true,
  supports_outgoing boolean NOT NULL DEFAULT true,
  incoming_auth text NOT NULL,
  outgoing_auth text NOT NULL,
  default_imap_host text,
  default_imap_port int,
  default_imap_secure boolean,
  default_smtp_host text,
  default_smtp_port int,
  default_smtp_secure boolean,
  help_text text,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

CREATE TABLE IF NOT EXISTS tenant_email_prefs (
  tenant_id text PRIMARY KEY,
  bcc_self boolean NOT NULL DEFAULT false,
  read_receipts boolean NOT NULL DEFAULT false,
  show_on_dashboard boolean NOT NULL DEFAULT true,
  contacts_only boolean NOT NULL DEFAULT true,
  updated_at timestamptz DEFAULT now()
);

Routes (tenant-scoped):

// server/src/routes/email-providers.ts
router.get("/email/providers", requireTenant, async (req, res) => {
  const providers = await storage.getEmailProviders();
  res.json({ ok: true, providers });
});

// dev-only seed
router.post("/email/providers/seed", async (req, res) => {
  await storage.seedEmailProviders(EMAIL_PROVIDER_SEED);
  res.json({ ok: true });
});

// prefs
router.get("/email/prefs", requireTenant, async (req: any, res) => {
  const prefs = await storage.getTenantEmailPrefs(req.tenantId);
  res.json({ ok: true, prefs });
});
router.put("/email/prefs", requireTenant, async (req: any, res) => {
  await storage.upsertTenantEmailPrefs(req.tenantId, req.body);
  res.json({ ok: true });
});
